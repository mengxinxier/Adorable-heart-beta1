	news_event = {
		id = china_warlord_news.001
		title = china_warlord_news.001.title
		picture = GFX_news_The_Snow_Mountain_Incident
		desc =  china_warlord_news.001.desc
		major = no
		is_triggered_only = yes
		immediate = {
			hidden_effect = {
				TIB = {
					remove_ideas = TIB_Tri_Trinity_Council_idea
					add_ideas = TIB_Enciodes_Silverash_idea
					set_party_name = { ideology = neutrality long_name = TIB_democratic_party name = TIB_democratic_party }
				}
			}
		}
		option = {
			name = china_warlord_news.001.a
			hidden_effect = {
				if = {
					limit = { ROOT = { tag = FROM } }
					every_other_country = {
						limit = {
							OR = {
								check_variable = { THIS.Chinese_Secondary_News_button_var = 2 }
								is_chinese_country = yes
								tag = ENG
							}
						}
						news_event = { id = china_warlord_news.001 }
					}
				}
			}
			trigger = {
				tag = TIB
			}
		}
		option = {
			name = china_warlord_news.001.b
			trigger = {
				is_chinese_country = yes
				not = { tag = TIB }
			}
		}
		option = {
			name = china_warlord_news.001.c
			trigger = {
				is_chinese_country = no
				not = { tag = ENG }
				not = { tag = TIB }
			}
		}
		option = {
			name = china_warlord_news.001.d
			trigger = {
				tag = ENG
			}
		}
	}
