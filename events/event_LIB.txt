			}
			send_equipment = {
				amount = 300
				target = FROM
				type = artillery_equipment
			}	
		}
		hidden_effect = {
			add_equipment_to_stockpile = {
				type = infantry_equipment
				amount = -2500
			}
			add_equipment_to_stockpile = {
				type = artillery_equipment
				amount = -300
			}
		}
		ai_chance = {
			base = 0
			modifier = {
				add = 1.0
				OR = {
					check_variable = {
						PRC.num_equipment@infantry_equipment < 0
					}
					check_variable = {
						PRC.num_equipment@artillery_equipment < 0
					}
				}
			}
			modifier = {
				add = 0.5
				FROM = {
					has_war = yes
				}
			}
			modifier = {
				factor = 100
				has_equipment = {
					infantry_equipment > 5000
				}
				has_equipment = {
					artillery_equipment > 500
				}
			}
		}
	}
	option = {
		name = LIB.2.b
		FROM = { country_event = LIB.4 }
		ai_chance = {
			base = 1
		}
	}
}
