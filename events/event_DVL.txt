	id = WizardCountry.005
	title = WizardCountry.005.titile
	desc = WizardCountry.005.desc
	picture = GFX_event_Minister_of_Economy
	is_triggered_only = yes
	option = {
		name = WizardCountry.005.a
		add_to_variable = {
			var = jobless_rate
			value = 1
			tooltip = jobless_rate_up_1_tt
		}
		add_to_variable = {
			var = undead_substitution_rate
			value = -1
			tooltip = undead_substitution_rate_down_1_tt

			var = undead_substitution_rate
			min = 0

		}
		ai_chance = {
			base = 10
		}
	}
}
