	id = shiyu.500
	title = shiyu.500.t
	desc = shiyu.500.d
	picture = GFX_report_event_canadian_soldiers
	is_triggered_only = no
	fire_only_once = no
	trigger = {
		SHY = {
			has_country_leader = {
				ruling_only = yes
				character = SHY_carlos_prio
			}
		}
		NOT = {
			has_country_flag = MGU_ANTI_SHY_MAGIC
		}
		OR = {
			AND = {
				has_war_with = SHY
				SHY = {
					has_idea = yidianmofa
				}
			}
			AND = {
				has_war_with = AST
				AST = {
					has_idea = yidianmofa
				}
			}
		}
		not = {
			has_idea = diren
		}
		not = {
			has_idea = chengfa
		}
	}
	option = {
		name = shiyu.500.b
		if = {
			limit = {
				has_war_with = SHY
			}
			add_ideas = diren
		}
		else_if = {
			limit = {
				has_war_with = AST
			}
			add_ideas = chengfa
		}
	}
}
