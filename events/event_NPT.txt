			trigger = { NOT = { has_country_flag = NPT_civil_war_between_three_sides_flag } }
		}

			hidden_effect = {

					NPT_civil_war_between_two_sides = yes

			hidden_effect = {
				MOR = { country_event = { id = neptune.040 days = 3 } }
