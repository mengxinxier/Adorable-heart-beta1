		visible = {
			always = yes

	Kivotos_check_robot_limit = {
		visible = {
			always = yes
		}
		activation = {
			always = yes
		}
		available = {
			IF = {
				limit = {
					#如果机器人超出库存
					has_idea = Kivotos_robot_limit_overflow
				}
				custom_trigger_tooltip = {
					#那么检测机器人数量是否小于等于上限值
					tooltip = Kivotos_check_robot_limit_trigger_tt_off
					set_temp_variable = {
						Kivotos_robot_num_temp = num_equipment@Kivotos_robot
					}
					divide_temp_variable = {
						Kivotos_robot_num_temp = 1000
					}
					check_variable = {
						var = Kivotos_robot_num_temp
						value = Kivotos_robot_limit
						compare = less_than_or_equals
					}
				}
				else = {
					custom_trigger_tooltip = {
						#反之则检测机器人数量是否大于库存上限
						tooltip = Kivotos_check_robot_limit_trigger_tt
						set_temp_variable = {
							Kivotos_robot_num_temp = num_equipment@Kivotos_robot
						}
						divide_temp_variable = {
							Kivotos_robot_num_temp = 1000
						}
						check_variable = {
							Kivotos_robot_num_temp > Kivotos_robot_limit
						}
					}
				}
			}
		}
		cancel_trigger = {
			always = no
		}
		days_mission_timeout = 3650
		timeout_effect = {
			hidden_effect = {
				#超时则自动重置
				activate_mission = Kivotos_check_robot_limit
			}
		}
		complete_effect = {
			if = {
				limit = {
					#如果超出库存，当满足条件1时则消除debuff
					has_idea = Kivotos_robot_limit_overflow
				}
				remove_ideas = Kivotos_robot_limit_overflow
				hidden_effect = {
					activate_mission = Kivotos_check_robot_limit
				}
				else = {
					#反之则附加限制生产debuff
					add_ideas = Kivotos_robot_limit_overflow
					hidden_effect = {
						activate_mission = Kivotos_check_robot_limit
					}
				}
			}
		}
	}

			base = 10

Kivotos_research_decision = {
	Kivotos_gain_research_point = {
		visible = {
			always = yes
		}
		activation = {
			always = yes
		}
		available = {
			always = no
		}
		cancel_trigger = {
			always = no
		}
		is_good = yes
		days_mission_timeout = 7
		timeout_effect = {
			set_temp_variable = {
				#获取科研点数修正值
				research_bonuses_temp = modifier@research_speed_factor
			}
			set_variable = {
				#获取科研点数每周基础值
				research_point_gain_temp = research_point_weekly
			}
			add_to_temp_variable = {
				#将100%加到修正值以得到最终修正
				research_bonuses_temp = 1
			}
			multiply_variable = {
				#将每周基础值进行修正
				research_point_gain_temp = research_bonuses_temp
			}
			add_to_variable = {
				#将最后得到数值加到固定变量
				research_point = research_point_gain_temp
			}
			if = {
				limit = {
					check_variable = {
						Kivotos_rank_up_num < 1
					}
					check_variable = {
						Kivotos_Tech_rank < 5
					}
				}
				add_to_variable = {
					Kivotos_Tech_rank = 1
				}
				add_to_variable = {
					Kivotos_rank_up_num = 10
				}
				if = {
					limit = {
						check_variable = {
							Kivotos_Tech_rank = 5
						}
					}
					set_variable = {
						Kivotos_rank_up_num = 0
					}
				}
			}
			Kivotos_student_army_num_check = yes
			hidden_effect = {
				#重置决议
				activate_mission = Kivotos_gain_research_point
			}
		}
	}
	Kivotos_research_solt_to_point = {
		visible = {
			always = yes
		}
		activation = {
			always = yes
		}
		available = {
			#如果科研槽大于0则触发完成效果
			amount_research_slots > 0
		}
		cancel_trigger = {
			always = no
		}
		days_mission_timeout = 3650
		timeout_effect = {
			#超时自动重置
			hidden_effect = {
				activate_mission = Kivotos_research_solt_to_point
			}
		}
		complete_effect = {
			#扣除1个科研槽，并获得200点每周科研点数
			add_research_slot = -1
			add_to_variable = {
				var = research_point_weekly
				value = 200
				tooltip = research_solt_to_point
			}
			hidden_effect = {
				activate_mission = Kivotos_research_solt_to_point
			}
		}
	}
