	##组建统一战线
	CHI_declare_Build_chinese_united_font = {
		allowed = {
		}
		visible = {
			any_country = {
				is_chinese_country = no
				has_war_with = ROOT
				is_subject = no
				is_major = yes
			}
			has_country_flag = Participate_in_the_competition_for_China_reunification
			NOT = { has_global_flag = CHI_REUNIT }
			or = {
				is_faction_leader = yes
				is_in_faction = no
			}
			if = {
				limit = {
					608 = {
						controller = {
							has_country_flag = Participate_in_the_competition_for_China_reunification
							or = {
								is_faction_leader = yes
								is_in_faction = no
							}
						}
					}
				}
				controls_state = 608
			}
		}
		available = {
			#controlled_states = 608
			is_subject = no
			not = {
				has_global_flag = CHINA_united_font_has_built
			}
			or = {
				is_faction_leader = yes
				is_in_faction = no
			}
		}
		fire_only_once = yes
		cost = 0
		ai_will_do = {
			factor = 100
		}
		complete_effect = {
			if = {
				limit = {
					not = { has_global_flag = The_Great_War_Against_Sakura }
					has_war_with = JAP
					JAP = { is_faction_leader = yes }
				}
				set_global_flag = The_Great_War_Against_Sakura
			}
			if = {
				limit = {
					NOT = {
						has_idea = CHI_protect_our_motherland
					}
				}
				add_ideas = CHI_protect_our_motherland
			}
			set_global_flag = CHINA_united_font_has_built
			set_country_flag = CHINA_united_font_leader
			if = {
				limit = {
					is_faction_leader = yes
				}
				set_country_flag = has_faction_before
				set_faction_name = chinese_united_font
			}
			else = {
				create_faction = chinese_united_font
			}
			every_other_country = {
				limit = {
					is_subject = no
					is_chinese_country = yes
				}
				country_event = {
					id = china_generic.001
				}
			}
			hidden_effect = {
				if = {
					limit = {
						CHI = {
							is_subject = no
						}
					}
					CHI = {
						if = {
							limit = {
								is_major = no
							}
							set_major = yes
						}
					}
				}
				if = {
					limit = {
						ROC = {
							is_subject = no
						}
					}
					ROC = {
						if = {
							limit = {
								is_major = no
							}
							set_major = yes
						}
					}
				}
				if = {
					limit = {
						PRC = {
							is_subject = no
						}
					}
					PRC = {
						if = {
							limit = {
								is_major = no
							}
							set_major = yes
						}
					}
				}
				news_event = {
					id = china_generic_news.001
					days = 1
				}
			}
		}
	}
