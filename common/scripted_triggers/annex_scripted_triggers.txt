	custom_trigger_tooltip = {
		tooltip = AH_is_an_ally
		exists = yes
		is_ally_with = ROOT
		NOT = {
			tag = ROOT
		}
	}

# in state scope
is_claim_or_core_of_ROOT = {
	OR = {
		is_claimed_by = ROOT
		is_core_of = ROOT
		ROOT = { has_variable = can_integrate_@PREV }
	}
}
