##军阀愿意臣服
CHI_warlords_are_willing_to_submit = {
	if = {
		limit = {
			ROOT = {
				has_country_flag = Participate_in_the_competition_for_China_reunification
				is_subject = no
			}
		}
		always = no
	}
	else_if = {
		limit = {
			ROOT = {
				tag = SIK
				has_country_leader = {
					ruling_only = yes
					character = CHI_Qiubai
				}
				is_subject = no
			}
		}
		always = yes
	}
	else_if = {
		limit = {
			ROOT = {
				tag = SIK
				has_country_leader = {
					ruling_only = yes
					character = CHI_Mejiro_Ardan
				}
				is_subject = no
			}
			FROM = {
				has_government = communism
			}
		}
		always = yes
	}
	else_if = {
		limit = {
			ROOT = {
				tag = TIB
				is_subject = no
			}
			FROM = {
				or = {
					has_government = communism
					has_government = democratic
				}
			}
		}
		always = yes
	}
	else_if = {
		limit = {
			ROOT = {
					has_government = communism
				is_subject = no
			}
			FROM = {
				has_government = communism
			}
		}
		always = yes
	}
	else = {
		always = no
	}
}
