on_actions = {
	on_peaceconference_ended = { 
		effect = { 
			if = {
				limit = { 
					ROOT = { tag = SOV }
					FROM = { NOT = { tag = SOV } }
					SOV = {
						NOT = { 
							any_enemy_country = {
								is_major = yes
						    } 
					    } 
						has_completed_focus = SOV_extraordinary_measures
					}
				}
				SOV = {
					country_event = SOV_flavor.7
				}
			}
		} 
	}
	
	on_state_control_changed = { 
		effect = {
			if = {
				limit = {
					SOV = {  has_country_flag = SOV_special_operation_failure_decsison }
					FROM = { tag = SOV }
					FROM.FROM = {
						is_core_of = SOV
						state = 219
					}
					ROOT = { tag = UKR }
				}
				SOV = {
					clr_country_flag = SOV_special_military_operations_start
				}
				UKR = {
					SOV = {
						country_event = SOV_end.1
					}
					country_event = { id = ukr_politics.9 days = 2 }
					remove_mission = UKR_a_vigorous_attack
					if = {
						limit = {
							any_enemy_country = {
								has_country_flag = UKR_intervene_in_war_soviet
							}
						}
						country_event = { id = rus_for.11 hours = 12 }
					}
				}
			}
			if = {
				limit = {
					FROM = { 
						tag = SOV
						has_war = yes
						is_subject = no
					}
					FROM.FROM = {
						OR = {
							state = 219
							state = 195
							state = 217
						}
					}
					SOV = {
						NOT = {
							any_enemy_country = {
								AND = {
									capital_scope = {
										OR = {
											is_on_continent = asia
											is_on_continent = india
											is_on_continent = australia
										}
									}
									is_major = yes
									is_subject = no
								}
							}
						}
						NOT = {
							has_country_flag = SOV_special_operation_failure_flag
						}
						NOT = {
							controls_state = 219
						}
						NOT = {
							controls_state = 195
						}
						NOT = {
							controls_state = 217
						}
					}
				}
				SOV = {
					country_event = SOV_end.1
				}
			}
			if = {
				limit = {
					SOV = {  has_country_flag = SOV_special_military_operations_start }
					FROM = { tag = UKR }
					FROM.FROM = {
						is_core_of = UKR
						state = 202
					}
					ROOT = {
						OR = {
							tag = SOV 
							is_in_faction_with = SOV
						}
					}
				}
				SOV = {
					annex_country = {
						target = UKR
					} 
					if = {
						limit = { has_country_flag = SOV_special_operation_failure_decsison }
						clr_country_flag = SOV_special_operation_failure_decsison
						country_event = { id = SOV_flavor.20 days = 1 }
					}
					clr_country_flag = SOV_special_military_operations_start
					country_event = SOV_flavor.23
					news_event = { id = SOV_news.7 hours = 6 }
				}
			}
		} 
	}

	on_weekly_SOV = { 
		effect = { 
			SOV = {
				SOV_flexible_economic_policy = yes
			}
		} 
	}

	on_daily_SOV = { 
		effect = { 
			if = {
				limit = { 
					SOV = { 
						any_enemy_country = { is_major = yes } 
						NOT = { has_war_with = UKR }
						is_ai = yes
						is_subject = no
						has_government = communism
					} 
				}
				if = {
					limit = {
						NOT = { 
							has_dynamic_modifier = {
								modifier = SOV_great_patriotic_war
							} 
						}
					}
					add_dynamic_modifier = {
						modifier = SOV_great_patriotic_war
					} 
				}
				SOV_great_patriotic_war_effect = yes
				if = {
					limit = {
						SOV = { has_war = no }
					}
					remove_dynamic_modifier = {
						modifier = SOV_great_patriotic_war
					} 
				}
			}
		} 
	}

	on_annex = {
		effect = {
			if = {
				limit = { FROM = { has_idea = SOV_comecon_buff } }
				FROM = { remove_ideas = SOV_comecon_buff }
			}
		}
	}

	on_war = {
		effect = {
			if = {
				limit = { 
					has_idea = SOV_comecon_buff 
					has_war_with = SOV
				}
				remove_ideas = SOV_comecon_buff
			}
		}
	}

	on_declare_war = {
		effect = {
			if = {
				limit = {
					ROOT = {
						tag = POL
						has_country_flag = POL_seek_roots_flag
					}
					FROM = { 
						tag = UKR
						has_war_with = SOV
						is_subject = no 
					}
				}
				POL = { 
					clr_country_flag = POL_seek_roots_flag
					country_event = polish.35 
				}
			}
		}
	}
	
}
