	on_declare_war = {
		effect = {
			if = {
				limit = {
					OR = {
						AND = {
							ROOT = {
								TAG = LAN
								has_country_flag = TUR_LAN_Middle_East_War_flag
							}
							FROM = {
								TAG = TUR
								has_country_flag = TUR_LAN_Middle_East_War_flag
							}
						}
						AND = {
							ROOT = {
								TAG = TUR
								has_country_flag = TUR_LAN_Middle_East_War_flag
							}
							FROM = {
								TAG = LAN
								has_country_flag = TUR_LAN_Middle_East_War_flag
							}
						}
					}
				}
				if = {
					limit = {
						ROOT = { TAG = LAN }
						FROM = { TAG = TUR }
					}
					hidden_effect = {
						if = {
							limit = { LAN = { NOT = { has_completed_focus = LAN_MEW_revange } } }
							LAN = { country_event = { id = saudi_arabia.59 days = 2 } }
						}
					}
				}
				if = {
					limit = {
						ROOT = { TAG = TUR }
						FROM = { TAG = LAN }
						PER = { is_in_faction_with = ROOT }
					}
					PER = {
						add_to_war = {
							targeted_alliance = ROOT
							enemy = LAN
							hostility_reason = asked_to_join
						}
					}
				}
				hidden_effect = {
					every_country = {
						limit = { is_ai = no }
						news_event =  { id = ME_news.2 hours = 2 }
					}
				}
			}
			if = {
				limit = {
					ROOT = {
						TAG = LAN
						has_completed_focus = MEOIL_expel_oil_company

			}

	on_war = {
		effect = {
			if = {
				limit = {
					original_tag = GRE
					has_idea = GRE_armed_forces_system_1
				}
				swap_ideas = {
					remove_idea = GRE_armed_forces_system_1
					add_idea = GRE_armed_forces_system_2
				}
			}
		}
	}
	
	on_peace = {
		effect = {
			if = {
				limit = {
					original_tag = GRE
					has_idea = GRE_armed_forces_system_2
				}
				swap_ideas = {
					remove_idea = GRE_armed_forces_system_2
					add_idea = GRE_armed_forces_system_1
				}
			}
		}
	}
	
	on_state_control_changed = {
		effect = {
			if = {
				limit = {
					FROM.FROM = { state = 266 }
					FROM = {
						has_war_with = ROOT
						has_country_flag = TUR_LAN_Middle_East_War_flag
					}
					ROOT = {
						has_war_with = FROM
						has_country_flag = TUR_LAN_Middle_East_War_flag
					}
				}
				FROM = {
					every_other_country = {
						limit = { has_war_with = PREV }
						FROM = {
							white_peace = {
								tag = PREV
								message = LAN_PER_MEW_war_message
							}
						}
					}
					leave_faction = yes
					every_core_state = {
						limit = { is_core_of = KUR }
						KUR = { transfer_state = PREV }
					}
				}
				if = {
					limit = {
						LAN = { is_faction_leader = yes }
						KUR = { NOT = { is_in_faction_with = LAN } }
						any_state = { is_controlled_by = KUR }
					}
					LAN = { add_to_faction = KUR }
				}
				if = {
					limit = {
						KUR = { NOT = { is_subject_of = LAN } }
						any_state = { is_controlled_by = KUR }
					}
					LAN = {
						set_autonomy = {
							target = KUR
							autonomous_state = autonomy_integrated_puppet
						}
					}
				}
				ROOT = {
					faction_leader = {
						country_event = { id = saudi_arabia.61 days = 1 }
					}
				}
			}
		}
	}
	
	on_offer_join_faction = {
		effect = {
			if = {
				limit = {
					FROM = {
						TAG = TUR
						OR = {
							has_completed_focus = TUR_join_eng
							has_completed_focus = TUR_union_across_sea
							has_completed_focus = TUR_trustworthy_alliance
							has_completed_focus = TUR_embryonic_form_of_strategic_resource_alliance
							has_completed_focus = TUR_associate_member_state
							has_war = yes
						}
						NOT = { is_in_faction_with = PER }
						faction_leader = { NOT = { has_war_with = PER } }
					}
				}
				FROM = {
					faction_leader = { PER = { country_event = generic.5 } }
				}
			}
		}
	}
	
	on_daily = {
		effect = {
			GRE = {

				}
			}
		}
	}

	on_civil_war_end = {
		effect = {
			if = {
				limit = {
					ROOT = { original_tag = TUR }
					FROM = { NOT = { original_tag = OTT } }
					all_enemy_country = { NOT = { has_country_flag = TUR_anatolian_civil_war_flag } }
				}
				clr_country_flag = TUR_anatolian_civil_war_flag
				country_lock_all_division_template = no
			}
			if = {
				limit = {
					ROOT = { original_tag = TUR }
					FROM = { original_tag = OTT }
				}
				TUR = { every_core_state = { remove_core_of = OTT } }
			}
		}
	}
