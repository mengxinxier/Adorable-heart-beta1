on_actions = {
	on_startup = {
		effect = {
			every_state = {
				if = {
					limit = {
						is_core_of = ENG
					}
					set_state_flag = EU_ENG_AREA
				}
				if = {
					limit = {
						is_core_of = NPT
					}
					set_state_flag = EU_NPT_AREA
				}
				if = {
					limit = {
						OR = {
							is_on_continent = europe
							AND = {
								is_on_continent = africa
								is_coastal = yes
							}
						}
						OR = {
							is_core_of = ITA
							is_core_of = MOW
							is_core_of = RMC
							is_core_of = ALS
						}
					}
					set_state_flag = EU_ITA_AREA
				}
				if = {
					limit = {
						OR = {
							is_core_of = TRB
							is_core_of = YUG
							is_core_of = GRE
							is_core_of = TUR
							is_core_of = BUL
						}
					}
					set_state_flag = EU_EAST_AREA
				}
				if = {
					limit = {
						is_core_of = SYR
						is_core_of = LEB
						is_core_of = PAL
						is_core_of = EGY
						is_core_of = LIB
					}
					set_state_flag = EU_SOUTH_AREA
				}
			}
		}
	}
	on_weekly = {
		effect = {
			if = {
				limit = {
					tag = ITA
					NOT = {
						has_country_flag = over3
					}
					OR = {
						has_global_flag = THE_BIG_DEPRESSION_has_come
						date>1936.2.20
					}
				}
				country_event = italia.22
				set_country_flag = over3
			}
			if = {
				limit = {
					NOT = {
						has_country_flag = over2
					}
					tag = ITA
					has_completed_focus = ITAo_1
					has_global_flag = THE_BIG_DEPRESSION_has_come
				}
				country_event = italia.2
				set_country_flag = over2
			}
		}
	}
	on_monthly = {
		effect = {
			if = {
				limit = {
					tag = ITA
					has_country_flag = nanahira1
					NOT = {
						has_country_flag = over1
					}
				}
				country_event = {
					days = 77
					id = nanahira_festival.4
				}
				set_country_flag = over1
			}
		}
	}
	on_capitulation = {
		effect = {
			if = {
				#意大利投降
				limit = {
					ROOT = {
						tag = ITA
						is_subject = no
						NOT = {
							any_enemy_country = {
								NOT = {
									OR = {
										tag = ITA
										tag = ALS
										tag = RMC
										tag = MOW
									}
								}
							}
						}
					}
					FROM = {
						OR = {
							tag = ITA
							tag = ALS
							tag = RMC
							tag = MOW
						}
					}
				}
				FROM = {
					white_peace = {
						tag = ITA
						message = ITA_on_capitulation
					}
					annex_country = {
						target = ITA
						transfer_troops = no
					}
					if = {
						limit = {
							MOW = {
								is_subject_of = ITA
							}
						}
						annex_country = {
							target = ITA
							transfer_troops = no
						}
					}
				}
			}
			if = {
				#那不勒斯
				limit = {
					ROOT = {
						tag = ALS
						is_subject = no
						NOT = {
							any_enemy_country = {
								NOT = {
									OR = {
										tag = ITA
										tag = ALS
										tag = RMC
										tag = MOW
									}
								}
							}
						}
					}
					FROM = {
						OR = {
							tag = ITA
							tag = ALS
							tag = RMC
							tag = MOW
						}
					}
				}
				FROM = {
					white_peace = {
						tag = ALS
						message = ALS_on_capitulation
					}
					annex_country = {
						target = ALS
						transfer_troops = no
					}
				}
			}
			if = {
				#
				limit = {
					ROOT = {
						tag = RMC
						is_subject = no
						NOT = {
							any_enemy_country = {
								NOT = {
									OR = {
										tag = ITA
										tag = ALS
										tag = RMC
										tag = MOW
									}
								}
							}
						}
					}
					FROM = {
						OR = {
							tag = ITA
							tag = ALS
							tag = RMC
							tag = MOW
						}
					}
				}
				FROM = {
					white_peace = {
						tag = RMC
						message = RMC_on_capitulation
					}
					annex_country = {
						target = RMC
						transfer_troops = no
					}
				}
			}
		}
	}
}
