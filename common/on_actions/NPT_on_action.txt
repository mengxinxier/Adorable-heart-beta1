on_action = {
	on_civil_war_end_before_annexation = {
		effect = {
			if = { #犯罪神寄了
				limit = {
					FROM = { tag = ARF }
					SPR_scw_in_progress = yes
					any_other_country = {
						NOT = {
							tag = ROOT
							tag = ARF
						}
						OR = {
							tag = BLA
							tag = NPT
						}
						exists = yes
					}
				}
				country_event = X
			}
			if = { #布兰寄了
				limit = {
					FROM = { tag = BLA }
					SPR_scw_in_progress = yes
					any_other_country = {
						NOT = {
							tag = ROOT
							tag = BLA
						}
						OR = {
							tag = ARF
							tag = NPT
						}
						exists = yes
					}
				}
				country_event = X
			}
			if = { #涅普姬雅寄了
			limit = {
				FROM = { tag = NPT }
				SPR_scw_in_progress = yes
				any_other_country = {
					NOT = {
						tag = ROOT
						tag = NPT
					}
					OR = {
						tag = ARF
						tag = BLA
					}
					exists = yes
				}
			}
			country_event = X
		}
		}
	}
}
