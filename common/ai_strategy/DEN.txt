		value = 200
	}

		type = front_control
		tag = SWE
		ratio > 0.10
		priority = 50
		ordertype = front
		execution_type = rush_weak
		manual_attack = no
	}
}

DEN_gain_volunteers_in_SWE = {

	allowed = { 
		OR = {
			original_tag = GER 
			original_tag = DVL
			original_tag = POL
		}
	}

	enable = {
		DEN = { has_war_with = SWE }
		if = {
			limit = { tag = DVL }
			SWE = {
				OR = {
					has_country_flag = SWE_Communist_Party_Cabinet
					has_government = communism
				}
			}
			DEN = {
				NOT = { 
					OR = {
						has_government = communism
						has_country_leader = {
							character = DEN_miho
							ruling_only = yes
						}
					}
				}
			}
		}
		if = {
			limit = { tag = GER }
			DEN = {
				has_country_leader = {
					character = DEN_miho
					ruling_only = yes
				}
			}
		}
		if = {
			limit = { tag = POL }
			is_in_faction_with = DVL
			SWE = {
				OR = {
					has_country_flag = SWE_Communist_Party_Cabinet
					has_government = communism
				}
			}
			DEN = {
				NOT = { 
					OR = {
						has_government = communism
						has_country_leader = {
							character = DEN_miho
							ruling_only = yes
						}
					}
				}
			}
		}
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = send_volunteers_desire
		id = DEN
		value = 100
	}

}
