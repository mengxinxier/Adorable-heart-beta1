scripted_gui = {
	DVL_Unit_Germany_Window = {
		window_name = "DVL_Unit_Germany_Window"
		visible = {
			tag = DVL
		}
		context_type = decision_category
		triggers = {
		}
		properties = {
			DVL_Unit_Germany_Button_KIL = { frame = DVL_Unit_Germany_Button_KIL_Var }
			DVL_Unit_Germany_Button_BAY = { frame = DVL_Unit_Germany_Button_BAY_Var }
			DVL_Unit_Germany_Button_RHI = { frame = DVL_Unit_Germany_Button_RHI_Var }
			Target_Country_flag_entry = { image = "[DVL_Target_Country.GetFlag]" }
			Target_Country_War_Progress = { frame = DVL_Target_Country.DVL_War_Progress_Var }
			Target_Country_Infiltrate_Progress = { frame = DVL_Target_Country.DVL_Infiltrate_Progress_Var }
		}
		effects = {
			DVL_Unit_Germany_Button_KIL_click = {
				set_variable = { DVL_Unit_Germany_Button_KIL_Var = 1 }
				set_variable = { DVL_Unit_Germany_Button_BAY_Var = 2 }
				set_variable = { DVL_Unit_Germany_Button_RHI_Var = 2 }
				SHL = { save_global_event_target_as = DVL_Target_Country }
			}
			DVL_Unit_Germany_Button_BAY_click = {
				set_variable = { DVL_Unit_Germany_Button_KIL_Var = 2 }
				set_variable = { DVL_Unit_Germany_Button_BAY_Var = 1 }
				set_variable = { DVL_Unit_Germany_Button_RHI_Var = 2 }
				BAY = { save_global_event_target_as = DVL_Target_Country }
			}
			DVL_Unit_Germany_Button_RHI_click = {
				set_variable = { DVL_Unit_Germany_Button_KIL_Var = 2 }
				set_variable = { DVL_Unit_Germany_Button_BAY_Var = 2 }
				set_variable = { DVL_Unit_Germany_Button_RHI_Var = 1 }
				RHI = { save_global_event_target_as = DVL_Target_Country }
			}
		}
	}
}
